Fizz-Buzz


                                        $("#btnFizzBuzz").on("click", function () {
                                            // get data
                                            var fizzNum = Number($("#fizzNum").val());
                                            var buzzNum = Number($("#buzzNum").val());
                                            var arrOutput = new Array();

                                            // do something with data
                                            for (var loop = 1; loop <= 100; loop++) {
                                                if ((loop % fizzNum == 0) && (loop % buzzNum == 0)) {
                                                    arrOutput.push("FIZZ-BUZZ");
                                                }
                                                else if (loop % fizzNum == 0) {
                                                    arrOutput.push("FIZZ");
                                                }
                                                else if (loop % buzzNum == 0) {
                                                    arrOutput.push("BUZZ");
                                                }
                                                else {
                                                    arrOutput.push(loop);
                                                }
                                            }

                                            // output
                                            $("#fizzOutput").html(arrOutput.join(", "));
                                        });

Taco Cat


                                        $("#btnTacoCat").on("click", function () {
                                            // change input data to string
                                            let word = String($("#txtTacoCat").val()).toLowerCase();

                                            // reverse word
                                            var revWord = "";

                                            for (var loop = word.length - 1; loop >= 0; loop--) {
                                                revWord += word[loop];
                                            }

                                            // output
                                            $("#tcOutput").text(`The reverse of ${word} is ${revWord} `);
                                            if (word === revWord)
                                                $("#tcOutput").append(`therefore ${word} is a Palidrome`);
                                            else
                                                $("#tcOutput").append(`therefore ${word} is NOT a Palidrome`);
                                        });

            myArray.push(Number($("#soafInput2").val()));

            myArray.push(Number($("#soafInput4").val()));
            myArray.push(Number($("#soafInput5").val()));

            let target = 0;
            let found = false;

            for (var index = 0; index < myArray.length; index++) {
                target = k - myArray[index];

                if (myArray.indexOf(target, index + 1) >= 0) {
                    found = true;
                    break;
                }
            }

            if (found)
                $("#soafOutput").html(`A K value of ${k} was found in the array`);
            else
                $("#soafOutput").html(`A K value of ${k} was NOT found in the array`);
        });

                                            function SunsetHills(buildings) {
                                            //Set tallest to the smallest possible number
                                            let tallest = Number.MIN_VALUE;

                                            //Here I want to build a new array that contains just the building that can see the sunset
                                            let sunsetViews = new Array();

                                            for (var index = 0; index < buildings.length; index++) {
                                                if (buildings[index] > tallest) {
                                                    tallest = buildings[index];
                                                    sunsetViews.push(`The Building in position ${index} with height ${tallest} can see the sunset. 
`);
                                                }
                                            }

                                            return sunsetViews.join("");
                                        };
